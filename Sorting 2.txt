class Solution {
    public int[] mergeSort(int[] nums) {
        if (nums == null || nums.length <= 1) {
            return nums;
        }
        divide(nums, 0, nums.length - 1);
        return nums;
    }

    private void divide(int[] nums, int low, int high) {
        if (low >= high) return;

        int mid = low + (high - low) / 2;

        // Recursively split the left and right halves
        divide(nums, low, mid);
        divide(nums, mid + 1, high);

        // Merge the sorted halves
        merge(nums, low, mid, high);
    }

    private void merge(int[] nums, int low, int mid, int high) {
        int[] temp = new int[high - low + 1];
        int left = low;
        int right = mid + 1;
        int k = 0;

        // Compare elements from both halves and copy the smaller one
        while (left <= mid && right <= high) {
            if (nums[left] <= nums[right]) {
                temp[k++] = nums[left++];
            } else {
                temp[k++] = nums[right++];
            }
        }

        // Copy remaining elements from the left half, if any
        while (left <= mid) {
            temp[k++] = nums[left++];
        }

        // Copy remaining elements from the right half, if any
        while (right <= high) {
            temp[k++] = nums[right++];
        }

        // Copy the sorted elements back into the original array
        for (int i = 0; i < temp.length; i++) {
            nums[low + i] = temp[i];
        }
    }
}

class Solution {
    public int[] bubbleSort(int[] nums) {
        // Start the recursive sorting from the full length of the array
        recursiveBubbleSort(nums, nums.length);
        return nums;
    }

    private void recursiveBubbleSort(int[] nums, int n) {
        // Base Case: If the size of the subarray is 1, it's already sorted
        if (n == 1) {
            return;
        }

        boolean swapped = false;

        // One pass of bubble sort to push the largest element to the end of nums[0...n-1]
        for (int i = 0; i < n - 1; i++) {
            if (nums[i] > nums[i + 1]) {
                // Swap nums[i] and nums[i+1]
                int temp = nums[i];
                nums[i] = nums[i + 1];
                nums[i + 1] = temp;
                swapped = true;
            }
        }

        // Optimization: If no elements were swapped, the array is already sorted
        if (!swapped) {
            return;
        }

        // Recursively sort the remaining n-1 elements
        recursiveBubbleSort(nums, n - 1);
    }
}


class Solution {
    public int[] insertionSort(int[] nums) {
        // Base case: An array of size 0 or 1 is already sorted
        if (nums == null || nums.length <= 1) {
            return nums;
        }
        
        // Start the recursive sorting from the last element
        sortRecursive(nums, nums.length);
        return nums;
    }
    
    // Recursive function to sort the array up to index 'n'
    private void sortRecursive(int[] nums, int n) {
        // Base case
        if (n <= 1) {
            return;
        }
        
        // Sort first n-1 elements
        sortRecursive(nums, n - 1);
        
        // Insert the last element at its correct position in sorted array
        int last = nums[n - 1];
        insertRecursive(nums, n - 2, last);
    }
    
    // Recursive function to insert 'last' into the sorted subarray nums[0...i]
    private void insertRecursive(int[] nums, int i, int last) {
        // Base case: if we reach the beginning or find an element smaller than/equal to 'last'
        if (i < 0 || nums[i] <= last) {
            nums[i + 1] = last;
            return;
        }
        
        // Shift element to the right
        nums[i + 1] = nums[i];
        
        // Check the next element to the left
        insertRecursive(nums, i - 1, last);
    }
}

class Solution {
    public int[] quickSort(int[] nums) {
        performQuickSort(nums, 0, nums.length - 1);
        return nums;
    }

    private void performQuickSort(int[] nums, int low, int high) {
        if (low < high) {
            // Find the partition index
            int pIndex = partition(nums, low, high);
            
            // Recursively sort elements before and after partition
            performQuickSort(nums, low, pIndex - 1);
            performQuickSort(nums, pIndex + 1, high);
        }
    }

    private int partition(int[] nums, int low, int high) {
        int pivot = nums[low]; // Take the first element as pivot
        int i = low;
        int j = high;

        while (i < j) {
            // Move i forward as long as elements are smaller than or equal to the pivot
            while (nums[i] <= pivot && i <= high - 1) {
                i++;
            }
            // Move j backward as long as elements are greater than the pivot
            while (nums[j] > pivot && j >= low + 1) {
                j--;
            }
            // Swap elements if pointers haven't crossed
            if (i < j) {
                swap(nums, i, j);
            }
        }
        // Place the pivot in its correct sorted position
        swap(nums, low, j);
        return j;
    }

    private void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
}
